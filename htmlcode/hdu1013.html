#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef long double ld;
const ld pi = acos(-1);
const int inf = 0x3f3f3f3f;
const ll llinf = 0x3f3f3f3f3f3f3f3f;

const int maxn = static_cast<int>(1e6 + 7);
struct node {
    int a, b, c, d;
    int idx;
    int sum;
} x[maxn];

bool cmp(const node &xx, const node &yy) {
    if (xx.sum == yy.sum) return xx.idx < yy.idx;
    return xx.sum > yy.sum;
}

int main() {
    ios::sync_with_stdio(false), cin.tie(nullptr);
    int n; cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> x[i].a >> x[i].b >> x[i].c >> x[i].d;
        x[i].idx = i;
        x[i].sum = x[i].a + x[i].b + x[i].c + x[i].d;
    }
    sort(x, x + n, cmp);
    for (int i = 0; i < n; i++) {
        if (x[i].idx == 0) {
            cout << i + 1 << endl;
            break;
        }
    }
}
