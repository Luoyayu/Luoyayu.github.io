funcbasename(sstring)string{
  fori:=len(s)-1;i>=0;i--{
     ifs[i]=='/'{
        s=s[i+1:]
        break
     }
   }
  fori:=len(s)-1;i>=0;i--{
     ifs[i]=='.'{
        s=s[:i]
        break
     }
   }
  returns
}

funcbasename2(sstring)string{
  slash:=strings.LastIndex(s,"/")
  s=s[slash+1:]

  ifdotIdx:=strings.LastIndex(s,".");dotIdx>=0{
     s=s[:dotIdx]
   }
  returns
}
