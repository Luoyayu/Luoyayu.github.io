typedef long long ll;

int main() {
    int n, k;
    string s;
    cin >> n >> k >> s;
    string ans;
    stack<int> stk;
    k >>= 1;
    for (int i = 0; i < n; i++) {
        if (s[i] == '(') {
            if (k) {
                k--;
                cout << "(";
                stk.push(1);
            } else {
                stk.push(0);
            }
        } else {
            if (stk.top()) cout << ")";
            stk.pop();
        }
    }
    cout << endl;
}
