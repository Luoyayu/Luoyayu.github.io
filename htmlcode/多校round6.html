#include <bits/stdc++.h>

using namespace std;

const int maxn = static_cast<int>(1e6 + 7);
double l, r;
const double esp = 1e-10;
double aa, bb;

double f(double t) { return 4 * aa * sqrt(1 - t * t / bb / bb) + 4 * t; }

inline double Simpson(double a, double b) {
    double MID = a + (b - a) / 2;
    return (f(a) + 4 * f(MID) + f(b)) * (b - a) / 6;
}

double ASR(double a, double b) {
    double sum = Simpson(a, b);
    double MID = a + (b - a) / 2;
    double suma = Simpson(a, MID);
    double sumb = Simpson(MID, b);
    return (fabs(sum - suma - sumb) < esp ? sum : ASR(a, MID) + ASR(MID, b));
}

int main() {
    int t; scanf("%d", &t);
    while (t--) {
        scanf("%lf %lf", &aa, &bb);
        l = 0, r = bb;
        double ans = ASR(0, bb) / bb;
        ans = (long long) (ans * 1e6);
        ans /= 1e6;
        printf("%.6f\n", ans);
    }
}
